1import torch, math 

2import torch.nn as nn 

3import torch.nn.functional as F 

4import numpy as np 

5 

6from LibMTL.architecture.MMoE import MMoE 

7 

8class DSelect_k(MMoE): 

9 r"""DSelect-k. 

10  

11 This method is proposed in `DSelect-k: Differentiable Selection in the Mixture of Experts with Applications to Multi-Task Learning (NeurIPS 2021) <https://proceedings.neurips.cc/paper/2021/hash/f5ac21cd0ef1b88e9848571aeb53551a-Abstract.html>`_ \ 

12 and implemented by modifying from the `official TensorFlow implementation <https://github.com/google-research/google-research/tree/master/dselect_k_moe>`_.  

13 

14 Args: 

15 img_size (list): The size of input data. For example, [3, 244, 244] denotes input images with size 3x224x224. 

16 num_experts (int): The number of experts shared by all the tasks. Each expert is an encoder network. 

17 num_nonzeros (int): The number of selected experts. 

18 kgamma (float, default=1.0): A scaling parameter for the smooth-step function. 

19 

20 """ 

21 def __init__(self, task_name, encoder_class, decoders, rep_grad, multi_input, device, **kwargs): 

22 super(DSelect_k, self).__init__(task_name, encoder_class, decoders, rep_grad, multi_input, device, **kwargs) 

23 

24 self._num_nonzeros = self.kwargs['num_nonzeros'] 

25 self._gamma = self.kwargs['kgamma'] 

26 

27 self._num_binary = math.ceil(math.log2(self.num_experts)) 

28 self._power_of_2 = (self.num_experts == 2 ** self._num_binary) 

29 

30 self._z_logits = nn.ModuleDict({task: nn.Linear(self.input_size, 

31 self._num_nonzeros*self._num_binary) for task in self.task_name}) 

32 self._w_logits = nn.ModuleDict({task: nn.Linear(self.input_size, self._num_nonzeros) for task in self.task_name}) 

33 

34 # initialization 

35 for param in self._z_logits.parameters(): 

36 param.data.uniform_(-self._gamma/100, self._gamma/100) 

37 for param in self._w_logits.parameters(): 

38 param.data.uniform_(-0.05, 0.05) 

39 

40 binary_matrix = np.array([list(np.binary_repr(val, width=self._num_binary)) \ 

41 for val in range(self.num_experts)]).astype(bool) 

42 self._binary_codes = torch.from_numpy(binary_matrix).to(self.device).unsqueeze(0) 

43 

44 self.gate_specific = None 

45 

46 def _smooth_step_fun(self, t, gamma=1.0): 

47 return torch.where(t<=-gamma/2, torch.zeros_like(t, device=t.device), 

48 torch.where(t>=gamma/2, torch.ones_like(t, device=t.device), 

49 (-2/(gamma**3))*(t**3) + (3/(2*gamma))*t + 1/2)) 

50 

51 def _entropy_reg_loss(self, inputs): 

52 loss = -(inputs*torch.log(inputs+1e-6)).sum() * 1e-6 

53 if not self._power_of_2: 

54 loss += (1/inputs.sum(-1)).sum() 

55 loss.backward(retain_graph=True) 

56 

57 def forward(self, inputs, task_name=None): 

58 experts_shared_rep = torch.stack([e(inputs) for e in self.experts_shared]) 

59 out = {} 

60 for task in self.task_name: 

61 if task_name is not None and task != task_name: 

62 continue 

63 sample_logits = self._z_logits[task](torch.flatten(inputs, start_dim=1)) 

64 sample_logits = sample_logits.reshape(-1, self._num_nonzeros, 1, self._num_binary) 

65 smooth_step_activations = self._smooth_step_fun(sample_logits) 

66 selector_outputs = torch.where(self._binary_codes.unsqueeze(0), smooth_step_activations, 

67 1 - smooth_step_activations).prod(3) 

68 selector_weights = F.softmax(self._w_logits[task](torch.flatten(inputs, start_dim=1)), dim=1) 

69 expert_weights = torch.einsum('ij, ij... -> i...', selector_weights, selector_outputs) 

70 gate_rep = torch.einsum('ij, ji... -> i...', expert_weights, experts_shared_rep) 

71 gate_rep = self._prepare_rep(gate_rep, task, same_rep=False) 

72 out[task] = self.decoders[task](gate_rep) 

73 

74 if self.training: 

75 self._entropy_reg_loss(selector_outputs) 

76 return out
