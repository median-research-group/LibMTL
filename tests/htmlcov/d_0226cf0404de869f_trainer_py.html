9class Trainer(nn.Module): 

10 r'''A Multi-Task Learning Trainer. 

11 

12 This is a unified and extensible training framework for multi-task learning.  

13 

14 Args: 

15 task_dict (dict): A dictionary of name-information pairs of type (:class:`str`, :class:`dict`). \ 

16 The sub-dictionary for each task has four entries whose keywords are named **metrics**, \ 

17 **metrics_fn**, **loss_fn**, **weight** and each of them corresponds to a :class:`list`. 

18 The list of **metrics** has ``m`` strings, repersenting the name of ``m`` metrics \ 

19 for this task. The list of **metrics_fn** has two elements, i.e., the updating and score \ 

20 functions, meaning how to update thoes objectives in the training process and obtain the final \ 

21 scores, respectively. The list of **loss_fn** has ``m`` loss functions corresponding to each \ 

22 metric. The list of **weight** has ``m`` binary integers corresponding to each \ 

23 metric, where ``1`` means the higher the score is, the better the performance, \ 

24 ``0`` means the opposite.  

25 weighting (class): A weighting strategy class based on :class:`LibMTL.weighting.abstract_weighting.AbsWeighting`. 

26 architecture (class): An architecture class based on :class:`LibMTL.architecture.abstract_arch.AbsArchitecture`. 

27 encoder_class (class): A neural network class. 

28 decoders (dict): A dictionary of name-decoder pairs of type (:class:`str`, :class:`torch.nn.Module`). 

29 rep_grad (bool): If ``True``, the gradient of the representation for each task can be computed. 

30 multi_input (bool): Is ``True`` if each task has its own input data, otherwise is ``False``.  

31 optim_param (dict): A dictionary of configurations for the optimizier. 

32 scheduler_param (dict): A dictionary of configurations for learning rate scheduler. \ 

33 Set it to ``None`` if you do not use a learning rate scheduler. 

34 kwargs (dict): A dictionary of hyperparameters of weighting and architecture methods. 

35 

36 .. note:: 

37 It is recommended to use :func:`LibMTL.config.prepare_args` to return the dictionaries of ``optim_param``, \ 

38 ``scheduler_param``, and ``kwargs``. 

39 

40 Examples::

131 def _process_data(self, loader): 

132 try: 

133 data, label = loader[1].next() 

134 except: 

135 loader[1] = iter(loader[0]) 

136 data, label = loader[1].next() 

137 data = data.to(self.device, non_blocking=True) 

138 if not self.multi_input: 

139 for task in self.task_name: 

140 label[task] = label[task].to(self.device, non_blocking=True) 

141 else: 

142 label = label.to(self.device, non_blocking=True) 

143 return data, label

159 if not self.multi_input: 

160 train_losses = torch.zeros(self.task_num).to(self.device)

179 def train(self, train_dataloaders, test_dataloaders, epochs, 

180 val_dataloaders=None, return_weight=False): 

181 r'''The training process of multi-task learning. 

182 

183 Args: 

184 train_dataloaders (dict or torch.utils.data.DataLoader): The dataloaders used for training. \ 

185 If ``multi_input`` is ``True``, it is a dictionary of name-dataloader pairs. \ 

186 Otherwise, it is a single dataloader which returns data and a dictionary \ 

187 of name-label pairs in each iteration. 

188 

189 test_dataloaders (dict or torch.utils.data.DataLoader): The dataloaders used for the validation or testing. \ 

190 The same structure with ``train_dataloaders``. 

191 epochs (int): The total training epochs. 

192 return_weight (bool): if ``True``, the loss weights will be returned. 

193 ''' 

194 train_loader, train_batch = self._prepare_dataloaders(train_dataloaders) 

195 train_batch = max(train_batch) if self.multi_input else train_batch 

196 

197 self.batch_weight = np.zeros([self.task_num, epochs, train_batch]) 

198 self.model.train_loss_buffer = np.zeros([self.task_num, epochs]) 

199 self.model.epochs = epochs 

200 for epoch in range(epochs): 

201 self.model.epoch = epoch 

202 self.model.train() 

203 self.meter.record_time('begin') 

204 for batch_index in range(train_batch): 

205 if batch_index >= 2: 

206 break 

207 if not self.multi_input: 

208 train_inputs, train_gts = self._process_data(train_loader) 

209 train_preds = self.model(train_inputs) 

210 train_preds = self.process_preds(train_preds) 

211 train_losses = self._compute_loss(train_preds, train_gts) 

212 self.meter.update(train_preds, train_gts) 

213 else: 

214 train_losses = torch.zeros(self.task_num).to(self.device) 

215 for tn, task in enumerate(self.task_name): 

216 train_input, train_gt = self._process_data(train_loader[task]) 

217 train_pred = self.model(train_input, task) 

218 train_pred = train_pred[task] 

219 train_pred = self.process_preds(train_pred, task) 

220 train_losses[tn] = self._compute_loss(train_pred, train_gt, task) 

221 self.meter.update(train_pred, train_gt, task) 

222 

223 self.optimizer.zero_grad() 

224 w = self.model.backward(train_losses, **self.kwargs['weight_args']) 

225 if w is not None: 

226 self.batch_weight[:, epoch, batch_index] = w 

227 self.optimizer.step() 

228 

229 self.meter.record_time('end') 

230 self.meter.get_score() 

231 self.model.train_loss_buffer[:, epoch] = self.meter.loss_item 

232 self.meter.display(epoch=epoch, mode='train') 

233 self.meter.reinit() 

234 

235 if val_dataloaders is not None: 

236 self.meter.has_val = True 

237 val_improvement = self.test(val_dataloaders, epoch, mode='val', return_improvement=True) 

238 self.test(test_dataloaders, epoch, mode='test') 

239 if self.scheduler is not None: 

240 if self.scheduler_param['scheduler'] == 'reduce' and val_dataloaders is not None: 

241 self.scheduler.step(val_improvement) 

242 else: 

243 self.scheduler.step() 

244 self.meter.display_best_result() 

245 if return_weight: 

246 return self.batch_weight

249 def test(self, test_dataloaders, epoch=None, mode='test', return_improvement=False): 

250 r'''The test process of multi-task learning. 

251 

252 Args: 

253 test_dataloaders (dict or torch.utils.data.DataLoader): If ``multi_input`` is ``True``, \ 

254 it is a dictionary of name-dataloader pairs. Otherwise, it is a single \ 

255 dataloader which returns data and a dictionary of name-label pairs in each iteration. 

256 epoch (int, default=None): The current epoch.  

257 ''' 

258 test_loader, test_batch = self._prepare_dataloaders(test_dataloaders) 

259 

260 self.model.eval() 

261 self.meter.record_time('begin') 

262 with torch.no_grad(): 

263 if not self.multi_input: 

264 for batch_index in range(test_batch): 

265 test_inputs, test_gts = self._process_data(test_loader) 

266 test_preds = self.model(test_inputs) 

267 test_preds = self.process_preds(test_preds) 

268 test_losses = self._compute_loss(test_preds, test_gts) 

269 self.meter.update(test_preds, test_gts) 

270 else: 

271 for tn, task in enumerate(self.task_name): 

272 for batch_index in range(test_batch[tn]): 

273 test_input, test_gt = self._process_data(test_loader[task]) 

274 test_pred = self.model(test_input, task)

277 test_loss = self._compute_loss(test_pred, test_gt, task) 

278 self.meter.update(test_pred, test_gt, task) 

279 self.meter.record_time('end') 

280 self.meter.get_score()
