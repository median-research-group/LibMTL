81class PLE(AbsArchitecture): 

82 r"""Progressive Layered Extraction (PLE). 

83  

84 This method is proposed in `Progressive Layered Extraction (PLE): A Novel Multi-Task Learning (MTL) Model for Personalized Recommendations (ACM RecSys 2020 Best Paper) <https://dl.acm.org/doi/10.1145/3383313.3412236>`_ \ 

85 and implemented by us.  

86 

87 Args: 

88 img_size (list): The size of input data. For example, [3, 244, 244] denotes input images with size 3x224x224. 

89 num_experts (list): The numbers of experts shared by all the tasks and specific to each task, respectively. Each expert is an encoder network. 

90 

91 .. warning:: 

92 - :class:`PLE` does not work with multi-input problems, i.e., ``multi_input`` must be ``False``. 

93 - :class:`PLE` is only supported by ResNet-based encoders. 

94 

95 """ 

96 def __init__(self, task_name, encoder_class, decoders, rep_grad, multi_input, device, **kwargs): 

97 super(PLE, self).__init__(task_name, encoder_class, decoders, rep_grad, multi_input, device, **kwargs) 

98 

99 if self.multi_input: 

100 raise ValueError('No support PLE for multiple inputs MTL problem') 

101 

102 self.img_size = self.kwargs['img_size'] 

103 self.input_size = np.array(self.img_size, dtype=int).prod() 

104 self.num_experts = {task: self.kwargs['num_experts'][tn+1] for tn, task in enumerate(self.task_name)} 

105 self.num_experts['share'] = self.kwargs['num_experts'][0] 

106 

107 self.encoder = {} 

108 for task in (['share']+self.task_name): 

109 self.encoder[task] = [self.encoder_class() for _ in range(self.num_experts[task])] 

110 self.encoder = _transform_resnet_PLE(self.encoder, task_name, self.img_size, 

111 self.num_experts, device)

113 def forward(self, inputs, task_name=None): 

114 out = {} 

115 gate_rep = self.encoder(inputs) 

116 for tn, task in enumerate(self.task_name): 

117 if task_name is not None and task != task_name: 

118 continue

120 ss_rep = self._prepare_rep(ss_rep, task, same_rep=False) 

121 out[task] = self.decoders[task](ss_rep)
