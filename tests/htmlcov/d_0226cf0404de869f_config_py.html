61def prepare_args(params): 

62 r"""Return the configuration of hyperparameters, optimizier, and learning rate scheduler. 

63 

64 Args: 

65 params (argparse.Namespace): The command-line arguments. 

66 """ 

67 kwargs = {'weight_args': {}, 'arch_args': {}} 

68 if params.weighting in ['EW', 'UW', 'GradNorm', 'GLS', 'RLW', 'MGDA', 'IMTL', 

69 'PCGrad', 'GradVac', 'CAGrad', 'GradDrop', 'DWA', 'Nash_MTL']: 

70 if params.weighting in ['DWA']: 

71 if params.T is not None: 

72 kwargs['weight_args']['T'] = params.T 

73 else: 

74 raise ValueError('DWA needs keywaord T') 

75 elif params.weighting in ['GradNorm']: 

76 if params.alpha is not None: 

77 kwargs['weight_args']['alpha'] = params.alpha 

78 else: 

79 raise ValueError('GradNorm needs keywaord alpha') 

80 elif params.weighting in ['MGDA']: 

81 if params.mgda_gn is not None: 

82 if params.mgda_gn in ['none', 'l2', 'loss', 'loss+']: 

83 kwargs['weight_args']['mgda_gn'] = params.mgda_gn 

84 else: 

85 raise ValueError('No support mgda_gn {} for MGDA'.format(params.mgda_gn)) 

86 else: 

87 raise ValueError('MGDA needs keywaord mgda_gn') 

88 elif params.weighting in ['GradVac']: 

89 if params.beta is not None: 

90 kwargs['weight_args']['beta'] = params.beta 

91 else: 

92 raise ValueError('GradVac needs keywaord beta') 

93 elif params.weighting in ['GradDrop']: 

94 if params.leak is not None: 

95 kwargs['weight_args']['leak'] = params.leak 

96 else: 

97 raise ValueError('GradDrop needs keywaord leak') 

98 elif params.weighting in ['CAGrad']: 

99 if params.calpha is not None and params.rescale is not None: 

100 kwargs['weight_args']['calpha'] = params.calpha 

101 kwargs['weight_args']['rescale'] = params.rescale 

102 else: 

103 raise ValueError('CAGrad needs keywaord calpha and rescale') 

104 elif params.weighting in ['Nash_MTL']: 

105 if params.update_weights_every is not None and params.optim_niter is not None and params.max_norm is not None: 

106 kwargs['weight_args']['update_weights_every'] = params.update_weights_every 

107 kwargs['weight_args']['optim_niter'] = params.optim_niter 

108 kwargs['weight_args']['max_norm'] = params.max_norm 

109 else: 

110 raise ValueError('Nash_MTL needs update_weights_every, optim_niter, and max_norm') 

111 else: 

112 raise ValueError('No support weighting method {}'.format(params.weighting)) 

113 

114 if params.arch in ['HPS', 'Cross_stitch', 'MTAN', 'CGC', 'PLE', 'MMoE', 'DSelect_k', 'DIY', 'LTB']: 

115 if params.arch in ['CGC', 'PLE', 'MMoE', 'DSelect_k']: 

116 kwargs['arch_args']['img_size'] = tuple(params.img_size)#np.array(params.img_size, dtype=int).prod() 

117 kwargs['arch_args']['num_experts'] = [int(num) for num in params.num_experts] 

118 if params.arch in ['DSelect_k']: 

119 kwargs['arch_args']['kgamma'] = params.kgamma 

120 kwargs['arch_args']['num_nonzeros'] = params.num_nonzeros 

121 else: 

122 raise ValueError('No support architecture method {}'.format(params.arch)) 

123 

124 if params.optim in ['adam', 'sgd', 'adagrad', 'rmsprop']: 

125 if params.optim == 'adam': 

126 optim_param = {'optim': 'adam', 'lr': params.lr, 'weight_decay': params.weight_decay} 

127 elif params.optim == 'sgd': 

128 optim_param = {'optim': 'sgd', 'lr': params.lr, 

129 'weight_decay': params.weight_decay, 'momentum': params.momentum} 

130 else: 

131 raise ValueError('No support optim method {}'.format(params.optim)) 

132 

133 if params.scheduler is not None: 

134 if params.scheduler in ['step', 'cos', 'exp']: 

135 if params.scheduler == 'step': 

136 scheduler_param = {'scheduler': 'step', 'step_size': params.step_size, 'gamma': params.gamma} 

137 else: 

138 raise ValueError('No support scheduler method {}'.format(params.scheduler)) 

139 else: 

140 scheduler_param = None 

141 

142 _display(params, kwargs, optim_param, scheduler_param) 

143 

144 return kwargs, optim_param, scheduler_param
